38 /* bmpread_t
39  *
40  * The struct filled by bmpread. Holds information about the image's pixels.
41  */
42 typedef struct bmpread_t
43 {
44  int width; /* width in pixels */
45  int height; /* height in pixels */
46 
47  /* A buffer holding the pixel data of the image. Each pixel spans three
48  * bytes: the red, green, and blue color components in that order. The
49  * pixels are ordered left to right, bottom line first (unless you passed
50  * BMPREAD_TOP_DOWN, in which case the top line is first). If the image has
51  * a width that's not divisible by 4, each line is padded at the end with
52  * unused bytes until its effective width is divisible by 4 (this behavior
53  * can be turned off by passing BMPREAD_BYTE_ALIGN).
54  */
55  unsigned char * rgb_data;
56 
57 } bmpread_t;

73 /* bmpread
74  *
75  * Loads the specified bitmap file from disk and fills out a bmpread_t struct
76  * with data about it.
77  *
78  * Inputs:
79  * bmp_file - The filename of the bitmap file to load.
80  * flags - One or more BMPREAD_* flags (defined above), combined with bitwise
81  * or. Specify 0 for standard, OpenGL compliant behavior.
82  * p_bmp_out - Pointer to a bmpread_t struct to fill with information.
83  *
84  * Returns:
85  * 0 if there's an error (file doesn't exist or is invalid, i/o error, etc.),
86  * or nonzero if the file loaded ok.
87  *
88  * Notes:
89  * The file must be a Windows format bitmap file, 1, 4, 8, or 24 bits, and not
90  * compressed (no RLE).
91  *
92  * Default behavior is for bmpread to return rgb_data in a format directly
93  * usable by OpenGL texture functions (e.g. glTexImage2D, format GL_RGB, type
94  * GL_UNSIGNED_BYTE). This implies a few oddities:
95  * * Lines are ordered bottom-first. To return data starting with the top
96  * line like you might otherwise expect, pass BMPREAD_TOP_DOWN in flags.
97  * * Lines are padded to be divisible by 4. To return data with no padding,
98  * pass BMPREAD_BYTE_ALIGN in flags.
99  * * Images with width or height that aren't an integer power of 2 will fail
100  * to load. To allow loading images of arbitrary size, pass
101  * BMPREAD_ANY_SIZE in flags.
102  * Note that passing any flags may cause the image to be unusable as an OpenGL
103  * texture, which may or may not matter to you.
104  */
105 int bmpread(const char * bmp_file, int flags, bmpread_t * p_bmp_out);
